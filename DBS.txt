CREATE TABLE Benutzer(
    Vorname VARCHAR(20),
    Nachname VARCHAR(20),
    Passwort VARCHAR(10),
    EMail VARCHAR(15),
    Username VARCHAR(7),
    Profilbild BLOB,
    Bildexist BOOLEAN,
    Adminrechte BOOLEAN,
    PRIMARY KEY(Username)
    );
    
CREATE TABLE platzreservierung(
 	ReservationID INTEGER,
 	PlaceID INTEGER,
 	RepeatID INTEGER,
	Username VARCHAR(7),
    Datum DATE,
    Zeit VARCHAR(13),
    Zeitstempel Timestamp,
    PRIMARY KEY(ReservationID),
    FOREIGN KEY (Username) REFERENCES Benutzer(Username)
	);
  
CREATE TABLE Kontakt(
	KontaktID SERIAL,
    Betreff TEXT,
    Nachricht TEXT,
    Name VARCHAR(25),
    EMail VARCHAR(30),
    Zeitstempel Timestamp,
    Status BOOLEAN,
	PRIMARY KEY(KontaktID)
    );
